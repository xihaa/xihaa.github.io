      ],
      listConfigParams: {
        tableInfo: {
          tableColumn: [
            {
              title: this.$t('global_version'),
              key: 'name'
            },
            {
              title: 'Lang',
              key: 'lang',
              align: 'center',
              renderHeader: (h, params) => {
                return h(
                  'Select',
                  {
                    props: {
                      value: this.lang,
                      size: 'small'

                    on: {
                      'on-change': e => {
                        this.lang = e;
                        this.changeLanguage();
                      }
                    }

                      props: {
                        value: item.label,
                        label: item.name
                      }
                    });
                  })

              }
            }
          ]
        }
      }
    };
  },
